# Supported Scopes: title
# Needs var:StA_title_allegiance to be set
StA_remove_current_title_allegiance = {
	if = {
		limit = {
			has_variable = StA_title_allegiance
			var:StA_title_allegiance = { has_variable = StA_title_realm_law }
		}
		var:StA_title_allegiance.var:StA_title_realm_law = {
			remove_list_variable = {
				name = StA_pledge_allegiance
				target = PREV
			}
		}
	}
	remove_variable = StA_title_allegiance
}

# Supported Scopes: title
# Holder needs to be a vassal
StA_set_title_allegiance_to_liege = {
	save_scope_as = title
	ordered_this_title_or_de_jure_above = {
		limit = {
			holder = $LIEGE$
			has_variable = StA_title_realm_law
		}
		order_by = reverse_tier
		var:StA_title_realm_law = {
			add_to_variable_list = {
				name = StA_pledge_allegiance
				target = scope:title
			}
		}
		scope:title = {
			set_variable = {
				name = StA_title_allegiance
				value = PREV
			}
		}
	}
	if = {
		limit = {
			NOT = { has_variable = StA_title_allegiance }
			holder = { is_independent_ruler = no }
			$LIEGE$.primary_title = { has_variable = StA_title_realm_law }
		}
		$LIEGE$.primary_title.var:StA_title_realm_law = {
			add_to_variable_list = {
				name = StA_pledge_allegiance
				target = scope:title
			}
		}
		set_variable = {
			name = StA_title_allegiance
			value = $LIEGE$.primary_title
		}
	}
	holder = {
		save_opinion_value_as = {
			name = opinion_liege
			target = liege
		}
		set_variable = {
			name = opinion_liege
			value = scope:opinion_liege
		}
	}
}

# Supported Scopes: title
StA_remove_allegiance_from_non_primary_titles = {
	holder = {
		every_held_title = {
			limit = {
				NOT = { THIS = PREV.primary_title }
				has_variable = StA_title_allegiance
			}
			StA_remove_current_title_allegiance = yes
		}
	}
}

StA_calculate_average_submission_ruler = {
  save_scope_as = ruler
  save_opinion_value_as = {
	name = opinion_liege
	target = liege
}
set_variable = {
	name = opinion_liege
	value = scope:opinion_liege
}
every_vassal = {
	save_opinion_value_as = {
		name = opinion_liege
		target = liege
	}
	set_variable = {
		name = opinion_liege
		value = scope:opinion_liege
	}
}
  every_held_title = {
    limit = { has_variable = StA_title_realm_law }
    var:StA_title_realm_law = {
      set_variable = {
        name = total_strength
        value = {
          every_in_list = {
            variable = StA_pledge_allegiance
            limit = {
              tier >= tier_county
              NOT = { holder = scope:ruler }
              THIS = holder.primary_title
            }
            add = holder.max_military_strength
          }
          min = 1
        }
      }
      set_variable = {
        name = average_submission
        value = {
          every_in_list = {
            variable = StA_pledge_allegiance
            limit = {
              tier >= tier_county
              NOT = { holder = scope:ruler }
              THIS = holder.primary_title
            }
            add = {
              add = holder.StA_submission
              multiply = holder.max_military_strength
            }
          }
          divide = var:total_strength
          max = 100
          min = -100
        }
      }
    }
  }
}
