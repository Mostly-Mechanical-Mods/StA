StA_update_cooldown_law_change = {
	if = {
		limit = { has_variable = StA_title_realm_law }

			if = {

				change_variable = {
					name = StA_cooldown_law_change

					name = StA_cooldown_law_change
					value = var:title.StA_cooldown_law_change
				}
			}
		}
	}
}

StA_reset_authority = {
	save_scope_as = story
	if = {
		limit = {
			debug_only = yes
			NOT = { has_variable = feudal_government_taxes_ }
		}
		debug_log = "StA_reset_authority NO GOVERNMENT TAXES"
		var:ruler = { debug_log_scopes = no }
		var:title = { debug_log_scopes = no }
	}
	var:ruler = {
		save_opinion_value_as = {
			name = opinion_liege
			target = liege
		}
		set_variable = {
			name = opinion_liege
			value = scope:opinion_liege
		}
		every_vassal = {
			save_opinion_value_as = {
				name = opinion_liege
				target = liege
			}
			set_variable = {
				name = opinion_liege
				value = scope:opinion_liege
			}
		}
	}
	set_variable = {
		name = total_strength
		value = {
			var:title = {
				every_in_list = {
					variable = StA_pledge_allegiance
					limit = {
						tier >= tier_county
						NOT = { holder = scope:story.var:title.holder }
						THIS = holder.primary_title
					}
					add = holder.max_military_strength
				}
				min = 1
			}
		}
	}
	set_variable = {
		name = average_submission
		value = {
			var:title = {
				every_in_list = {
					variable = StA_pledge_allegiance
					limit = {
						tier >= tier_county
						NOT = { holder = scope:story.var:title.holder }
						THIS = holder.primary_title
					}
					add = {
						add = holder.StA_submission
						multiply = holder.max_military_strength
					}
				}
			}
			divide = var:total_strength
			max = 100
			min = -100
		}
	}
	set_variable = {
		name = StA_title_realm_authority
		value = var:title.StA_title_realm_authority
	}
	set_variable = {
		name = StA_title_realm_authority_capacity
		value = var:title.StA_title_realm_authority_capacity
	}
}

set_succession_law_on_dependent_titles = {
	hidden_effect = {
		if = {
			limit = { debug_only = yes }
			debug_log = "PASS SINGLE HEIR LAW"
			debug_log_scopes = yes
		}
		if = {
			limit = { exists = scope:title }
			scope:title = {
				if = {
					limit = { debug_only = yes }
					debug_log_scopes = no
				}
				every_in_de_jure_hierarchy = {
					limit = {
						holder = PREV.holder
						is_leased_out = no
						NOR = {
							has_variable = StA_title_realm_law
							any_this_title_or_de_jure_above = {
								NOR = {
									THIS = PREV
									THIS = scope:title
									target_is_de_jure_liege_or_above = scope:title
								}
								holder = PREV.holder
								has_variable = StA_title_realm_law
							}
						}
					}
					if = {
						limit = { debug_only = yes }
						debug_log = "INHERITS LAW"
						debug_log_scopes = no
					}
					add_title_law = $LAW$
				}
			}
		}
	}
}
